# 文件搜寻

## VPATH

``` makefile
# 按照从左到右查找

VPATH := src:temp

TARGET := main
CC := gcc

$(TARGET) : main.o test.o
    $(CC) main.o test.o -o $(TARGET)

main.o : main.c
    $(CC) -c main.c

test.o : test.c
    $(CC) -c ./src/test.c

.PHONY : clean
clean:
    rm -f *.o $(TARGET)
```

## vpath

``` makefile
# 按照从上到下查找

vpath %.c src
vpath %.c temp

TARGET := main
CC := gcc

$(TARGET) : main.o test.o
    $(CC) main.o test.o -o $(TARGET)

main.o : main.c
    $(CC) -c main.c

test.o : test.c
    $(CC) -c ./src/test.c

.PHONY : clean
clean:
    rm -f *.o $(TARGET)
```
